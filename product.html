/* Chỉ cập nhật 2 hàm này trong Script của ông */ function openSmoothZoom(originImg) { if (animating) return; // Chặn click khi đang trong tiến trình animation if (activeOriginImg) forceCloseZoom(); animating = true; activeOriginImg = originImg; const rect = originImg.getBoundingClientRect(); const scrollBarWidth = window.innerWidth - document.documentElement.clientWidth; document.body.style.paddingRight = scrollBarWidth + "px"; document.body.classList.add("lock-scroll"); zoomImg.src = originImg.src; zoomImg.style.display = "block"; zoomImg.style.transition = "none"; zoomImg.style.width = rect.width + "px"; zoomImg.style.height = rect.height + "px"; zoomImg.style.transform = `translate3d(${rect.left}px,${rect.top}px,0)`; zoomImg.style.opacity = "0.5"; // Bắt đầu hơi mờ để tạo độ sâu zoomImg.offsetHeight; // Force reflow lightbox.classList.add("active"); originImg.style.visibility = "hidden"; requestAnimationFrame(() => { // Tăng tốc độ lên .45s để cảm giác dứt khoát hơn theo ý ông trước đó zoomImg.style.transition = "transform .45s cubic-bezier(.22,1,.36,1), width .45s cubic-bezier(.22,1,.36,1), height .45s cubic-bezier(.22,1,.36,1), opacity .45s ease"; const winW = window.innerWidth; const winH = window.innerHeight; const imgRatio = rect.width / rect.height; const winRatio = winW / winH; let targetW, targetH; if (imgRatio > winRatio) { targetW = winW * 0.92; targetH = targetW / imgRatio; } else { targetH = winH * 0.92; targetW = targetH * imgRatio; } const targetX = (winW - targetW) / 2; const targetY = (winH - targetH) / 2; zoomImg.style.width = targetW + "px"; zoomImg.style.height = targetH + "px"; zoomImg.style.transform = `translate3d(${targetX}px,${targetY}px,0)`; zoomImg.style.opacity = "1"; // Rõ nét hoàn toàn khi ra giữa // Mở khóa animating sau khi bung xong setTimeout(() => { animating = false; }, 450); }); } function closeSmoothZoom() { if (!activeOriginImg || animating) return; animating = true; const rect = activeOriginImg.getBoundingClientRect(); // Thu nhỏ kèm mờ dần zoomImg.style.transition = "transform .4s cubic-bezier(.22,1,.36,1), width .4s cubic-bezier(.22,1,.36,1), height .4s cubic-bezier(.22,1,.36,1), opacity .4s ease"; zoomImg.style.width = rect.width + "px"; zoomImg.style.height = rect.height + "px"; zoomImg.style.transform = `translate3d(${rect.left}px,${rect.top}px,0)`; zoomImg.style.opacity = "0"; // Mờ dần khi thu về lightbox.classList.remove("active"); setTimeout(() => { if (activeOriginImg) { activeOriginImg.style.visibility = "visible"; } zoomImg.style.display = "none"; document.body.classList.remove("lock-scroll"); document.body.style.paddingRight = ""; activeOriginImg = null; animating = false; }, 400); }
